
Apollo Query: GET ALL COURSES
------------------------------

query GetCourses {
    getCourses {
        id
        name
        slug
        courseCategory
        courseHistory
        courseIntro
        image {
            url
            public_id
        }
        modules {
            name
            description
            video {
                ETag
                ServerSideEncryption
                Location
                Key
                Bucket
            }
            image {
                url
                public_id
            }
        }
        instructor {
            name
            email
        }
    }
}
